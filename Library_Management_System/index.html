Library Management System (C)

A menu-driven program to manage a small library collection.

Project Summary

My project is a Library Management System developed in C language. It helps manage a small collection of books in a library. The system allows the user to add new books (with ID, title, and author), display all available books, search for a book by title, issue a book (mark it unavailable), and return a book (mark it available again).

Source Code

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Book {
    int id;
    char title[100];
    char author[100];
    int available;
};

int main() {
    struct Book books[100];
    int count = 0, choice, id, i;
    char title[100];

    while (1) {
        printf("\n=== Library Management System ===\n");
        printf("1. Add Book\n2. Display Books\n3. Search Book\n4. Issue Book\n5. Return Book\n6. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter Book ID: ");
                scanf("%d", &books[count].id);
                printf("Enter Book Title: ");
                getchar();
                gets(books[count].title);
                printf("Enter Author Name: ");
                gets(books[count].author);
                books[count].available = 1;
                count++;
                printf("Book added successfully!\n");
                break;

            case 2:
                printf("\n--- Book List ---\n");
                for (i = 0; i < count; i++) {
                    printf("ID: %d | Title: %s | Author: %s | %s\n",
                           books[i].id, books[i].title, books[i].author,
                           books[i].available ? "Available" : "Issued");
                }
                break;

            case 3:
                printf("Enter Book Title to search: ");
                getchar();
                gets(title);
                for (i = 0; i < count; i++) {
                    if (strcmp(books[i].title, title) == 0) {
                        printf("Book found - ID: %d | Author: %s | %s\n",
                               books[i].id, books[i].author,
                               books[i].available ? "Available" : "Issued");
                        break;
                    }
                }
                if (i == count) printf("Book not found.\n");
                break;

            case 4:
                printf("Enter Book ID to issue: ");
                scanf("%d", &id);
                for (i = 0; i < count; i++) {
                    if (books[i].id == id) {
                        if (books[i].available) {
                            books[i].available = 0;
                            printf("Book issued successfully!\n");
                        } else {
                            printf("Book already issued.\n");
                        }
                        break;
                    }
                }
                if (i == count) printf("Book not found.\n");
                break;

            case 5:
                printf("Enter Book ID to return: ");
                scanf("%d", &id);
                for (i = 0; i < count; i++) {
                    if (books[i].id == id) {
                        books[i].available = 1;
                        printf("Book returned successfully!\n");
                        break;
                    }
                }
                if (i == count) printf("Book not found.\n");
                break;

            case 6:
                printf("Exiting the program.\n");
                exit(0);

            default:
                printf("Invalid choice! Try again.\n");
        }
    }
    return 0;
}
